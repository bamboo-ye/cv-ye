	var err error
	once.Do(func() {
		config, err = loadConfig(configPath)
	})
	return config, err
}

func loadConfig(configPath string) (*Config, error) {
	v := viper.New()
	
	// 设置配置路径和名称
	if configPath != "" {
		v.SetConfigFile(configPath)
	} else {
		v.AddConfigPath("./configs")
		v.AddConfigPath(".")
		v.SetConfigName("config")
		v.SetConfigType("yaml")
	}
	
	// 设置默认值
	setDefaults(v)
	
	// 读取配置
	if err := v.ReadInConfig(); err != nil {
		return nil, fmt.Errorf("failed to read config: %w", err)
	}
	
	// 监听配置变化
	v.WatchConfig()
	v.OnConfigChange(func(e fsnotify.Event) {
		fmt.Printf("Config file changed: %s\n", e.Name)
	})
	
	// 环境变量覆盖
	v.AutomaticEnv()
	
	// 解析配置
	var cfg Config
	if err := v.Unmarshal(&cfg); err != nil {
		return nil, fmt.Errorf("failed to unmarshal config: %w", err)
	}
	
	return &cfg, nil
}

// JSONMap 用于存储JSON数据的map类型
type JSONMap map[string]interface{}

func (m JSONMap) Value() (driver.Value, error) {
	if m == nil {
		return nil, nil
	}
	return json.Marshal(m)
}

func (m *JSONMap) Scan(value interface{}) error {
	if value == nil {
		*m = nil
		return nil
	}
	
	bytes, ok := value.([]byte)
	if !ok {
		return errors.New("type assertion to []byte failed")
	}
	
	var result map[string]interface{}
	if err := json.Unmarshal(bytes, &result); err != nil {
		return err
	}
	*m = result
	return nil
}

package database

import (
	"fmt"
	"log"
	
	"gorm.io/gorm"
	
	"emoji-system/internal/model"

func Migrate(db *gorm.DB) error {
	log.Println("开始数据库迁移...")
	
	err := db.AutoMigrate(
		&model.User{},
		&model.EmojiTask{},
		&model.Template{},
		&model.Message{},
	)
	
	if err != nil {
		return fmt.Errorf("数据库迁移失败: %w", err)
	}
	
	// 创建索引
	createIndexes(db)
	
	// 初始化模板数据
	initTemplates(db)
	
	log.Println("数据库迁移完成")
	return nil
}

func createIndexes(db *gorm.DB) {
	// 为EmojiTask表创建额外索引
	db.Exec("CREATE INDEX IF NOT EXISTS idx_task_user_status ON emoji_tasks(user_id, status)")
	db.Exec("CREATE INDEX IF NOT EXISTS idx_task_created ON emoji_tasks(created_at)")
	
	// 为Message表创建额外索引
	db.Exec("CREATE INDEX IF NOT EXISTS idx_message_expired ON messages(expired_at) WHERE expired_at IS NOT NULL")
}

// NewTaskDispatcher 创建任务分发器
func NewTaskDispatcher(

// CreateTask 创建表情包生成任务
func (d *TaskDispatcher) CreateTask(ctx context.Context, req TaskRequest) (*TaskResponse, error) {
	startTime := time.Now()
	
	// 1. 验证用户
	if err := d.validateUser(ctx, req.UserID); err != nil {
		return nil, err
	}
	
	// 2. 验证模板
	template, err := d.validateTemplate(ctx, req.TemplateID)
	if err != nil {
		return nil, err
	}
	
	// 3. 验证文本长度
	if len(req.TextContent) > template.MaxTextLength {
		return nil, errors.NewValidationError(
			fmt.Sprintf("文本长度不能超过%d个字符", template.MaxTextLength),
		)
	}
	
	// 4. 生成任务ID
	taskID := utils.GenerateTaskID()
	
	// 5. 创建数据库记录
	task := &model.EmojiTask{
		ID:               taskID,
		UserID:           req.UserID,
		TemplateID:       req.TemplateID,
		TextContent:      req.TextContent,
		Style:            req.Style,
		Status:           model.TaskStatusPending,
		Priority:         req.Priority,
		Metadata:         req.Metadata,
		CreatedAt:        time.Now(),
		UpdatedAt:        time.Now(),
	}
	
	if err := d.db.Create(task).Error; err != nil {
		d.logger.Error("创建任务数据库记录失败",
			zap.Error(err),
			zap.String("task_id", taskID),
			zap.Uint64("user_id", req.UserID),
		)
		return nil, errors.NewDatabaseError("创建任务失败")
	}
	
	// 6. 发送到Kafka
	kafkaMsg := &kafka.TaskMessage{
		TaskID:      taskID,
		UserID:      req.UserID,
		TemplateID:  req.TemplateID,
		TextContent: req.TextContent,
		Style:       req.Style,
		Priority:    int(req.Priority),
		CreatedAt:   time.Now().Unix(),
		Metadata:    req.Metadata,
	}
	
	if err := d.sendToKafka(ctx, kafkaMsg); err != nil {
		// 发送失败，更新任务状态
		d.db.Model(task).Updates(map[string]interface{}{
			"status":         model.TaskStatusFailed,
			"error_message":  "任务队列发送失败",
			"updated_at":     time.Now(),
		})
		return nil, errors.NewServiceError("系统繁忙，请稍后重试")
	}
	
	// 7. 缓存任务信息
	if err := d.cacheTask(ctx, taskID, task); err != nil {
		d.logger.Warn("缓存任务信息失败",
			zap.Error(err),
			zap.String("task_id", taskID),
		)
	}
	
	// 8. 计算预计等待时间
	waitTime := d.calculateEstimatedWaitTime()
	
	// 9. 更新指标
	d.metrics.IncTaskCreated()
	d.metrics.ObserveTaskCreationTime(time.Since(startTime))
	
	d.logger.Info("任务创建成功",
		zap.String("task_id", taskID),
		zap.Uint64("user_id", req.UserID),
		zap.String("template_id", req.TemplateID),
		zap.Int("estimated_wait_time", waitTime),
	)
	
	return &TaskResponse{
		TaskID:            taskID,
		Status:            "pending",
		EstimatedWaitTime: waitTime,
		CreatedAt:         task.CreatedAt,
	}, nil
}

// GetTaskStatus 获取任务状态
func (d *TaskDispatcher) GetTaskStatus(ctx context.Context, taskID string, userID uint64) (*TaskStatusResponse, error) {
	// 1. 尝试从缓存获取
	if status, err := d.getTaskFromCache(ctx, taskID); err == nil {
		// 检查用户权限
		if status.UserID != userID {
			return nil, errors.NewPermissionError("无权访问此任务")
		}
		return d.convertToStatusResponse(status), nil
	}
	
	// 2. 从数据库获取
	task, err := d.getTaskFromDB(ctx, taskID)
	if err != nil {
		return nil, err
	}
	
	// 检查用户权限
	if task.UserID != userID {
		return nil, errors.NewPermissionError("无权访问此任务")
	}
	
	// 3. 更新缓存
	go d.cacheTask(context.Background(), taskID, task)
	
	return d.convertToStatusResponse(task), nil
}

// CancelTask 取消任务
func (d *TaskDispatcher) CancelTask(ctx context.Context, taskID string, userID uint64) error {
	task, err := d.getTaskFromDB(ctx, taskID)
	if err != nil {
		return err
	}
	
	// 检查用户权限
	if task.UserID != userID {
		return errors.NewPermissionError("无权取消此任务")
	}
	
	// 检查任务状态
	if task.Status != model.TaskStatusPending && task.Status != model.TaskStatusProcessing {
		return errors.NewBusinessError("任务无法取消，当前状态: " + task.Status.String())
	}
	
	// 更新数据库
	updates := map[string]interface{}{
		"status":     model.TaskStatusCancelled,
		"updated_at": time.Now(),
	}
	
	if err := d.db.Model(&model.EmojiTask{}).
		Where("id = ?", taskID).
		Updates(updates).Error; err != nil {
		return errors.NewDatabaseError("取消任务失败")
	}
	
	// 更新缓存
	go d.updateTaskCache(context.Background(), taskID, map[string]interface{}{
		"status":     model.TaskStatusCancelled,
		"updated_at": time.Now().Unix(),
	})
	
	// 发送取消消息到Kafka
	cancelMsg := &kafka.CancelMessage{
		TaskID:    taskID,
		UserID:    userID,
		Timestamp: time.Now().Unix(),
	}
	
	d.sendCancelToKafka(ctx, cancelMsg)
	
	d.logger.Info("任务取消成功",
		zap.String("task_id", taskID),
		zap.Uint64("user_id", userID),
	)
	
	return nil
}

// BatchGetTaskStatus 批量获取任务状态
func (d *TaskDispatcher) BatchGetTaskStatus(ctx context.Context, taskIDs []string, userID uint64) (map[string]*TaskStatusResponse, error) {
	result := make(map[string]*TaskStatusResponse)
	
	// 批量从缓存获取
	cachedTasks := d.batchGetTasksFromCache(ctx, taskIDs)
	
	// 检查哪些任务需要从数据库获取
	var needDBFetch []string
	for _, taskID := range taskIDs {
		if task, ok := cachedTasks[taskID]; ok {
			if task.UserID == userID {
				result[taskID] = d.convertToStatusResponse(task)
			}
		} else {
			needDBFetch = append(needDBFetch, taskID)
		}
	}
	
	// 从数据库获取剩余任务
	if len(needDBFetch) > 0 {
		dbTasks, err := d.batchGetTasksFromDB(ctx, needDBFetch)
		if err != nil {
			return nil, err
		}
		
		for _, task := range dbTasks {
			if task.UserID == userID {
				result[task.ID] = d.convertToStatusResponse(task)
				// 异步更新缓存
				go d.cacheTask(context.Background(), task.ID, task)
			}
		}
	}
	
	return result, nil
}

func (d *TaskDispatcher) getTaskFromCache(ctx context.Context, taskID string) (*model.EmojiTask, error) {
	key := fmt.Sprintf("task:%s", taskID)
	data, err := d.cache.HGetAll(ctx, key)
	if err != nil {
		return nil, err
	}
	
	if len(data) == 0 {
		return nil, fmt.Errorf("任务不存在于缓存中")
	}
	
	task := &model.EmojiTask{
		ID:           data["task_id"].(string),
		UserID:       uint64(data["user_id"].(int64)),
		TemplateID:   data["template_id"].(string),
		Status:       model.TaskStatus(data["status"].(int64)),
		ResultURL:    data["result_url"].(string),
		ErrorMessage: data["error_msg"].(string),
		CreatedAt:    time.Unix(data["created_at"].(int64), 0),
		UpdatedAt:    time.Unix(data["updated_at"].(int64), 0),
	}
	
	if completedAt, ok := data["completed_at"].(int64); ok && completedAt > 0 {
		completedTime := time.Unix(completedAt, 0)
		task.CompletedAt = &completedTime
	}
	
	return task, nil
}

func (d *TaskDispatcher) calculateEstimatedWaitTime() int {
	// 获取队列中的任务数量
	queueSize := d.metrics.GetQueueSize()
	
	// 获取当前处理速度
	processingSpeed := d.metrics.GetProcessingSpeed()
	
	if processingSpeed == 0 {
		processingSpeed = 10 // 默认每秒处理10个任务
	}
	
	// 计算预计等待时间（秒）
	waitTime := queueSize / processingSpeed
	
	// 加上基础处理时间
	waitTime += 5
	
	// 限制最大等待时间
	if waitTime > 300 {
		waitTime = 300
	}
	
	return waitTime
}

func (d *TaskDispatcher) convertToStatusResponse(task *model.EmojiTask) *TaskStatusResponse {
	resp := &TaskStatusResponse{
		TaskID:       task.ID,
		Status:       task.Status.String(),
		ResultURL:    task.ResultURL,
		ErrorMessage: task.ErrorMessage,
		CreatedAt:    task.CreatedAt,
		UpdatedAt:    task.UpdatedAt,
		CompletedAt:  task.CompletedAt,
	}
	
	// 计算进度
	switch task.Status {
	case model.TaskStatusPending:
		resp.Progress = 0
	case model.TaskStatusProcessing:
		resp.Progress = 50
	case model.TaskStatusSuccess:
		resp.Progress = 100
	case model.TaskStatusFailed, model.TaskStatusCancelled:
		resp.Progress = 100
	}
	
	return resp
}

type RedisCache struct {

// RedisConfig Redis配置
type RedisConfig struct {

	ReadTimeout  time.Duration

// NewRedisCache 创建Redis缓存
func NewRedisCache(config *RedisConfig, logger *zap.Logger) (*RedisCache, error) {
	rdb := redis.NewClusterClient(&redis.ClusterOptions{
		Addrs:        config.Addrs,
		Password:     config.Password,
		PoolSize:     config.PoolSize,
		MinIdleConns: config.MinIdleConns,
		DialTimeout:  config.DialTimeout,
		ReadTimeout:  config.ReadTimeout,
		WriteTimeout: config.WriteTimeout,
		MaxRetries:   config.MaxRetries,
	})
	
	// 测试连接
	ctx, cancel := context.WithTimeout(context.Background(), 5*time.Second)
	defer cancel()
	
	if err := rdb.Ping(ctx).Err(); err != nil {
		return nil, fmt.Errorf("Redis连接失败: %w", err)
	}
	
	cache := &RedisCache{
		client: rdb,
		logger: logger,
		config: config,
	}
	
	logger.Info("Redis缓存初始化成功",
		zap.Strings("addrs", config.Addrs),
		zap.Int("pool_size", config.PoolSize),
	)
	
	return cache, nil
}

// Get 获取缓存
func (c *RedisCache) Get(ctx context.Context, key string) (string, error) {
	startTime := time.Now()
	
	value, err := c.client.Get(ctx, key).Result()
	if err != nil {
		if err == redis.Nil {
			c.logger.Debug("缓存未命中", zap.String("key", key))
			return "", ErrCacheMiss
		}
		c.logger.Error("获取缓存失败",
			zap.String("key", key),
			zap.Error(err),
		)
		return "", fmt.Errorf("获取缓存失败: %w", err)
	}
	
	c.logger.Debug("缓存命中",
		zap.String("key", key),
		zap.Duration("latency", time.Since(startTime)),
	)
	
	return value, nil
}

// Set 设置缓存
func (c *RedisCache) Set(ctx context.Context, key string, value interface{}, expiration time.Duration) error {
	var val string
	switch v := value.(type) {
	case string:
		val = v
	case []byte:
		val = string(v)
	default:
		jsonBytes, err := json.Marshal(value)
		if err != nil {
			return fmt.Errorf("序列化缓存值失败: %w", err)
		}
		val = string(jsonBytes)
	}

// Start 启动消费者
func (c *AlgorithmConsumer) Start(ctx context.Context) error {
	c.mu.Lock()
	if c.isRunning {
		c.mu.Unlock()
		return fmt.Errorf("消费者已经在运行")
	}
	c.isRunning = true
	c.mu.Unlock()
	
	c.logger.Info("启动算法消费者")
	
	// 启动指标收集
	go c.collectMetrics(ctx)
	
	// 启动健康检查
	go c.healthCheck(ctx)
	
	// 启动消费循环
	for {
		select {
		case <-ctx.Done():
			c.logger.Info("收到停止信号，停止消费者")
			return c.Stop()
		default:
			// 检查负载情况
			if !c.shouldConsume() {
				time.Sleep(1 * time.Second)
				continue
			}
			
			// 开始消费
			handler := &consumerGroupHandler{
				consumer: c,
				ctx:      ctx,
			}
			
			err := c.consumerGroup.Consume(ctx, []string{c.config.Topic}, handler)
			if err != nil {
				c.logger.Error("消费失败",
					zap.Error(err),
				)
				
				// 等待一段时间后重试
				select {
				case <-ctx.Done():
					return nil
				case <-time.After(5 * time.Second):
				}
			}
		}
	}
}

func (h *consumerGroupHandler) ConsumeClaim(
	session sarama.ConsumerGroupSession,
	claim sarama.ConsumerGroupClaim,
) error {
	h.consumer.logger.Info("开始消费分区",
		zap.String("topic", claim.Topic()),
		zap.Int32("partition", claim.Partition()),
		zap.Int64("initial_offset", claim.InitialOffset()),
	)
	
	batch := make([]*sarama.ConsumerMessage, 0, h.consumer.config.BatchSize)
	batchTimer := time.NewTimer(h.consumer.config.PollTimeout)
	
	for {
		select {
		case <-h.ctx.Done():
			h.consumer.logger.Info("收到停止信号，停止消费")
			return nil
			
		case message, ok := <-claim.Messages():
			if !ok {
				h.consumer.logger.Info("消息通道已关闭")
				return nil
			}
			
			// 检查是否应该处理消息
			if !h.consumer.shouldProcessMessage(message) {
				h.consumer.logger.Debug("跳过消息处理",
					zap.String("task_id", string(message.Key)),
					zap.String("reason", "负载过高或限流"),
				)
				continue
			}
			
			// 添加到批次
			batch = append(batch, message)
			
			// 检查批次是否已满
			if len(batch) >= h.consumer.config.BatchSize {
				h.processBatch(session, batch)
				batch = batch[:0]
				batchTimer.Reset(h.consumer.config.PollTimeout)
			}
			
		case <-batchTimer.C:
			// 处理当前批次
			if len(batch) > 0 {
				h.processBatch(session, batch)
				batch = batch[:0]
			}
			batchTimer.Reset(h.consumer.config.PollTimeout)
		}
	}
}

// processBatch 处理消息批次
func (h *consumerGroupHandler) processBatch(
	session sarama.ConsumerGroupSession,
	messages []*sarama.ConsumerMessage,
) {
	if len(messages) == 0 {
		return
	}
	
	h.consumer.logger.Debug("处理消息批次",
		zap.Int("batch_size", len(messages)),
	)
	
	// 使用工作池处理消息
	var wg sync.WaitGroup
	semaphore := make(chan struct{}, h.consumer.concurrencyLimit)
	
	for _, message := range messages {
		wg.Add(1)
		
		go func(msg *sarama.ConsumerMessage) {
			defer wg.Done()
			
			// 获取信号量（控制并发）
			semaphore <- struct{}{}
			defer func() { <-semaphore }()
			
			// 处理单条消息
			h.processMessage(session, msg)
		}(message)
	}
	
	wg.Wait()
	
	// 批量提交偏移量
	for _, message := range messages {
		session.MarkMessage(message, "")
	}
	
	h.consumer.metrics.MessagesConsumed += int64(len(messages))
}

// processMessage 处理单条消息
func (h *consumerGroupHandler) processMessage(
	session sarama.ConsumerGroupSession,
	message *sarama.ConsumerMessage,
) {
	startTime := time.Now()
	taskID := string(message.Key)
	
	h.consumer.logger.Debug("开始处理消息",
		zap.String("task_id", taskID),
		zap.Int64("offset", message.Offset),
		zap.Int32("partition", message.Partition),
	)
	
	// 解析消息
	var taskMsg kafka.TaskMessage
	if err := json.Unmarshal(message.Value, &taskMsg); err != nil {
		h.consumer.logger.Error("解析消息失败",
			zap.String("task_id", taskID),
			zap.Error(err),
		)
		h.consumer.metrics.MessagesFailed++
		return
	}
	
	// 检查消息类型
	messageType := getMessageType(message.Headers)
	if messageType == "cancel" {
		h.handleCancelMessage(taskMsg.TaskID)
		return
	}
	
	// 检查任务状态
	if h.isTaskCancelled(taskMsg.TaskID) {
		h.consumer.logger.Debug("任务已取消，跳过处理",
			zap.String("task_id", taskMsg.TaskID),
		)
		return
	}
	
	// 更新任务状态为处理中
	if err := h.updateTaskStatus(taskMsg.TaskID, model.TaskStatusProcessing); err != nil {
		h.consumer.logger.Error("更新任务状态失败",
			zap.String("task_id", taskMsg.TaskID),
			zap.Error(err),
		)
		return
	}
	
	// 创建处理上下文
	ctx, cancel := context.WithTimeout(h.ctx, 30*time.Second)
	defer cancel()
	
	// 记录任务开始处理
	h.consumer.recordTaskStart(taskMsg.TaskID, cancel)
	
	// 调用算法接口生成表情包
	result, err := h.consumer.algorithmAPI.GenerateEmoji(ctx,
		taskMsg.TemplateID,
		taskMsg.TextContent,
		taskMsg.Style,
		taskMsg.Metadata,
	)
	
	processingTime := int(time.Since(startTime).Milliseconds())
	
	// 处理结果
	if err != nil {
		h.consumer.logger.Error("算法处理失败",
			zap.String("task_id", taskMsg.TaskID),
			zap.Error(err),
			zap.Int("processing_time_ms", processingTime),
		)
		
		// 更新任务为失败状态
		h.updateTaskStatus(taskMsg.TaskID, model.TaskStatusFailed, map[string]interface{}{
			"error_message":    err.Error(),
			"processing_time":  processingTime,
		})
		
		h.consumer.metrics.MessagesFailed++
	} else {
		h.consumer.logger.Info("算法处理成功",
			zap.String("task_id", taskMsg.TaskID),
			zap.String("result_url", result.ImageURL),
			zap.Int("processing_time_ms", processingTime),
		)
		
		// 更新任务为成功状态
		h.updateTaskStatus(taskMsg.TaskID, model.TaskStatusSuccess, map[string]interface{}{
			"result_url":       result.ImageURL,
			"processing_time":  processingTime,
			"metadata":         result.Metadata,
		})
		
		h.consumer.metrics.MessagesProcessed++
	}
	
	// 发送推送消息
	h.sendPushNotification(taskMsg.TaskID, taskMsg.UserID, err == nil, result)
	
	// 记录任务处理完成
	h.consumer.recordTaskComplete(taskMsg.TaskID)
	
	// 记录处理时间
	h.consumer.recordProcessingTime(processingTime)
}

// 辅助方法
func (c *AlgorithmConsumer) shouldConsume() bool {
	// 检查负载监控
	strategy := c.monitor.GetCurrentStrategy()
	if strategy.Level > 0 {
		// 系统处于降级状态，减少消费
		c.concurrencyLimit = strategy.MaxConcurrency
		c.rateLimiter.SetLimit(strategy.MaxConcurrency)
	}
	
	// 检查并发限制
	c.mu.RLock()
	current := c.currentConcurrent
	c.mu.RUnlock()
	
	return current < c.concurrencyLimit
}

func (c *AlgorithmConsumer) shouldProcessMessage(message *sarama.ConsumerMessage) bool {
	// 检查限流器
	if !c.rateLimiter.Allow() {
		return false
	}
	
	// 检查消息优先级
	priority := getMessagePriority(message.Headers)
	
	// 在系统负载高时，只处理高优先级消息
	strategy := c.monitor.GetCurrentStrategy()
	if strategy.Level > 1 && priority < 3 {
		return false
	}
	
	return true
}

func (c *AlgorithmConsumer) recordTaskStart(taskID string, cancel context.CancelFunc) {
	c.mu.Lock()
	c.processingTasks[taskID] = cancel
	c.currentConcurrent++
	c.mu.Unlock()
	
	c.metrics.CurrentConcurrency = c.currentConcurrent
}

func (c *AlgorithmConsumer) recordTaskComplete(taskID string) {
	c.mu.Lock()
	delete(c.processingTasks, taskID)
	c.currentConcurrent--
	c.mu.Unlock()
	
	c.metrics.CurrentConcurrency = c.currentConcurrent
}

func (c *AlgorithmConsumer) recordProcessingTime(processingTime int) {
	// 更新平均处理时间
	c.metrics.AvgProcessingTime = (c.metrics.AvgProcessingTime*float64(c.metrics.MessagesProcessed-1) + 
		float64(processingTime)) / float64(c.metrics.MessagesProcessed)
}

func (c *AlgorithmConsumer) updateTaskStatus(taskID string, status model.TaskStatus, updates ...map[string]interface{}) error {
	dbUpdates := map[string]interface{}{
		"status":     status,
		"updated_at": time.Now(),
	}
	
	if status == model.TaskStatusSuccess || status == model.TaskStatusFailed {
		now := time.Now()
		dbUpdates["completed_at"] = now
	}
	
	// 合并额外更新
	if len(updates) > 0 {
		for k, v := range updates[0] {
			dbUpdates[k] = v
		}
	}
	
	// 更新数据库

		Where("id = ?", taskID).
		Updates(dbUpdates).Error; err != nil {

	}
	
	// 更新缓存
	go c.updateTaskCache(taskID, dbUpdates)
	
	return nil
}

	}
}

// PushToUser 推送消息给用户
func (s *Service) PushToUser(ctx context.Context, userID uint64, message *Message) error {
	startTime := time.Now()
	
	// 设置消息ID和时间戳
	if message.ID == "" {
		message.ID = generateMessageID()
	}
	if message.Timestamp == 0 {
		message.Timestamp = time.Now().Unix()
	}
	
	// 检查用户是否在线
	if s.isUserOnline(userID) {
		// 实时推送
		if err := s.pushRealTime(ctx, userID, message); err != nil {
			s.logger.Warn("实时推送失败，转为离线推送",
				zap.Uint64("user_id", userID),
				zap.String("message_id", message.ID),
				zap.Error(err),
			)
			// 失败后转为离线推送
			return s.pushOffline(ctx, userID, message)
		}
		
		s.metrics.RealTimePushes++
		s.metrics.PushSuccess++
		s.recordPushLatency(time.Since(startTime))
		
		s.logger.Debug("实时推送成功",
			zap.Uint64("user_id", userID),
			zap.String("message_id", message.ID),
			zap.String("message_type", string(message.Type)),
		)
	} else {
		// 离线推送
		if err := s.pushOffline(ctx, userID, message); err != nil {
			s.metrics.PushFailed++
			return err
		}
		
		s.metrics.OfflinePushes++
		s.metrics.PushSuccess++
		s.recordPushLatency(time.Since(startTime))
		
		s.logger.Debug("离线推送成功",
			zap.Uint64("user_id", userID),
			zap.String("message_id", message.ID),
			zap.String("message_type", string(message.Type)),
		)
	}
	
	s.metrics.TotalPushes++
	
	return nil
}

// BatchPushToUsers 批量推送消息给多个用户
func (s *Service) BatchPushToUsers(ctx context.Context, userMessages map[uint64]*Message) error {
	if len(userMessages) == 0 {
		return nil
	}
	
	startTime := time.Now()
	
	var wg sync.WaitGroup
	errors := make(chan error, len(userMessages))
	semaphore := make(chan struct{}, 100) // 限制并发数
	
	for userID, message := range userMessages {
		wg.Add(1)
		
		go func(uid uint64, msg *Message) {
			defer wg.Done()
			
			semaphore <- struct{}{}
			defer func() { <-semaphore }()
			
			if err := s.PushToUser(ctx, uid, msg); err != nil {
				errors <- fmt.Errorf("用户 %d 推送失败: %w", uid, err)
			}
		}(userID, message)
	}
	
	wg.Wait()
	close(errors)
	
	// 收集错误
	var errs []error
	for err := range errors {
		errs = append(errs, err)
	}
	
	s.logger.Debug("批量推送完成",
		zap.Int("total_users", len(userMessages)),
		zap.Int("failed_count", len(errs)),
		zap.Duration("duration", time.Since(startTime)),
	)
	
	if len(errs) > 0 {
		return fmt.Errorf("批量推送完成，%d 个用户推送失败", len(errs))
	}
	
	return nil
}

// PushToAllOnline 推送给所有在线用户
func (s *Service) PushToAllOnline(ctx context.Context, message *Message) error {
	s.onlineUsersMu.RLock()
	userIDs := make([]uint64, 0, len(s.onlineUsers))
	for userID := range s.onlineUsers {
		userIDs = append(userIDs, userID)
	}
	s.onlineUsersMu.RUnlock()
	
	if len(userIDs) == 0 {
		return nil
	}
	
	userMessages := make(map[uint64]*Message)
	for _, userID := range userIDs {
		// 复制消息
		msgCopy := *message
		msgCopy.ID = generateMessageID()
		userMessages[userID] = &msgCopy
	}
	
	return s.BatchPushToUsers(ctx, userMessages)
}

// RegisterUserOnline 注册用户在线
func (s *Service) RegisterUserOnline(userID uint64, session *UserSession) error {
	s.onlineUsersMu.Lock()
	defer s.onlineUsersMu.Unlock()
	
	session.LastSeen = time.Now()
	s.onlineUsers[userID] = session
	
	// 更新Redis中的在线状态
	ctx := context.Background()
	onlineKey := fmt.Sprintf("user:online:%d", userID)
	if err := s.redis.Set(ctx, onlineKey, "1", 5*time.Minute).Err(); err != nil {
		s.logger.Error("更新用户在线状态失败",
			zap.Uint64("user_id", userID),
			zap.Error(err),
		)
	}
	
	// 发送离线消息
	go s.deliverOfflineMessages(userID, session.WsConnection)
	
	s.logger.Info("用户上线",
		zap.Uint64("user_id", userID),
		zap.String("device_id", session.DeviceID),
	)
	
	return nil
}

// UnregisterUserOnline 注销用户在线
func (s *Service) UnregisterUserOnline(userID uint64) {
	s.onlineUsersMu.Lock()
	defer s.onlineUsersMu.Unlock()
	
	if _, exists := s.onlineUsers[userID]; exists {
		delete(s.onlineUsers, userID)
		
		// 清理Redis中的在线状态
		ctx := context.Background()
		onlineKey := fmt.Sprintf("user:online:%d", userID)
		s.redis.Del(ctx, onlineKey)
		
		s.logger.Info("用户下线",
			zap.Uint64("user_id", userID),
		)
	}
}

// 内部方法
func (s *Service) isUserOnline(userID uint64) bool {
	// 先检查内存中的在线状态
	s.onlineUsersMu.RLock()
	_, exists := s.onlineUsers[userID]
	s.onlineUsersMu.RUnlock()
	
	if exists {
		return true
	}
	
	// 检查Redis中的在线状态（用于跨实例检查）
	ctx := context.Background()
	onlineKey := fmt.Sprintf("user:online:%d", userID)
	exists, err := s.redis.Exists(ctx, onlineKey).Result()
	if err != nil {
		s.logger.Error("检查用户在线状态失败",
			zap.Uint64("user_id", userID),
			zap.Error(err),
		)
		return false
	}
	
	return exists > 0
}

func (s *Service) pushRealTime(ctx context.Context, userID uint64, message *Message) error {
	// 获取用户会话
	session, exists := s.GetUserSession(userID)
	if !exists {
		return fmt.Errorf("用户会话不存在")
	}
	
	// 检查WebSocket连接是否有效
	if session.WsConnection == nil || !session.WsConnection.IsConnected() {
		return fmt.Errorf("WebSocket连接无效")
	}
	
	// 序列化消息
	messageBytes, err := json.Marshal(message)
	if err != nil {
		return fmt.Errorf("序列化消息失败: %w", err)
	}
	
	// 发送消息
	if err := session.WsConnection.Send(messageBytes); err != nil {
		// 发送失败，标记用户离线
		s.UnregisterUserOnline(userID)
		return fmt.Errorf("发送消息失败: %w", err)
	}
	
	// 更新用户最后活跃时间
	s.UpdateUserLastSeen(userID)
	
	return nil
}

func (s *Service) pushOffline(ctx context.Context, userID uint64, message *Message) error {
	// 使用Redis Stream存储离线消息
	streamKey := fmt.Sprintf("offline:messages:%d", userID)
	
	messageData, err := json.Marshal(message)
	if err != nil {
		return fmt.Errorf("序列化消息失败: %w", err)
	}
	
	args := &redis.XAddArgs{
		Stream: streamKey,
		Values: map[string]interface{}{
			"message": string(messageData),
			"type":    message.Type,
			"time":    time.Now().Unix(),
		},
	}
	
	// 添加到Stream
	_, err = s.redis.XAdd(ctx, args).Result()
	if err != nil {
		return fmt.Errorf("添加离线消息失败: %w", err)
	}
	
	// 限制Stream长度
	s.redis.XTrimMaxLen(ctx, streamKey, int64(s.config.MaxOfflineMessages))
	
	// 设置过期时间
	s.redis.Expire(ctx, streamKey, s.config.OfflineMessageTTL)
	
	// 更新内存中的离线队列（用于快速访问）
	s.offlineQueuesMu.Lock()
	if _, exists := s.offlineQueues[userID]; !exists {
		s.offlineQueues[userID] = []*Message{}
	}
	s.offlineQueues[userID] = append(s.offlineQueues[userID], message)
	s.offlineQueuesMu.Unlock()
	
	return nil
}

func (s *Service) deliverOfflineMessages(userID uint64, wsConn *WebSocketConnection) {
	ctx := context.Background()
	
	// 从Redis Stream获取离线消息
	streamKey := fmt.Sprintf("offline:messages:%d", userID)
	
	messages, err := s.redis.XRange(ctx, streamKey, "-", "+").Result()
	if err != nil {
		s.logger.Error("获取离线消息失败",
			zap.Uint64("user_id", userID),
			zap.Error(err),
		)
		return
	}
	
	if len(messages) == 0 {
		return
	}
	
	s.logger.Info("开始投递离线消息",
		zap.Uint64("user_id", userID),
		zap.Int("message_count", len(messages)),
	)
	
	var delivered []string
	for _, msg := range messages {
		// 解析消息
		messageData := msg.Values["message"].(string)
		var message Message
		if err := json.Unmarshal([]byte(messageData), &message); err != nil {
			s.logger.Error("解析离线消息失败",
				zap.String("message_id", msg.ID),
				zap.Error(err),
			)
			continue
		}
		
		// 发送消息
		messageBytes, _ := json.Marshal(message)
		if err := wsConn.Send(messageBytes); err != nil {
			s.logger.Error("发送离线消息失败",
				zap.String("message_id", msg.ID),
				zap.Error(err),
			)
			break
		}
		
		delivered = append(delivered, msg.ID)
		
		s.logger.Debug("离线消息投递成功",
			zap.String("message_id", msg.ID),
			zap.String("type", string(message.Type)),
		)
		
		// 短暂延迟，避免对客户端造成压力
		time.Sleep(50 * time.Millisecond)
	}
	
	// 删除已投递的消息
	if len(delivered) > 0 {
		// 使用XDEL删除消息
		s.redis.XDel(ctx, streamKey, delivered...)
		
		// 清理内存中的离线队列
		s.offlineQueuesMu.Lock()
		if len(delivered) == len(s.offlineQueues[userID]) {
			delete(s.offlineQueues, userID)
		} else {
			// 移除已投递的消息
			remaining := make([]*Message, 0, len(s.offlineQueues[userID])-len(delivered))
			for i, msg := range s.offlineQueues[userID] {
				found := false
				for _, deliveredID := range delivered {
					if msg.ID == deliveredID {
						found = true
						break
					}
				}
				if !found {
					remaining = append(remaining, s.offlineQueues[userID][i])
				}
			}
			s.offlineQueues[userID] = remaining
		}
		s.offlineQueuesMu.Unlock()
		
		s.logger.Info("离线消息投递完成",
			zap.Uint64("user_id", userID),
			zap.Int("delivered_count", len(delivered)),
		)
	}
}

func (s *Service) startCleanupWorker() {
	ticker := time.NewTicker(5 * time.Minute)
	defer ticker.Stop()
	
	for range ticker.C {
		s.cleanupInactiveSessions()
		s.cleanupExpiredMessages()
	}
}

func (s *Service) cleanupInactiveSessions() {
	s.onlineUsersMu.Lock()
	defer s.onlineUsersMu.Unlock()
	
	inactiveThreshold := time.Now().Add(-10 * time.Minute)
	var inactiveUsers []uint64
	
	for userID, session := range s.onlineUsers {
		if session.LastSeen.Before(inactiveThreshold) {
			inactiveUsers = append(inactiveUsers, userID)
		}
	}
	
	for _, userID := range inactiveUsers {
		delete(s.onlineUsers, userID)
		
		// 清理Redis中的在线状态
		ctx := context.Background()
		onlineKey := fmt.Sprintf("user:online:%d", userID)
		s.redis.Del(ctx, onlineKey)
		
		s.logger.Info("清理非活跃会话",
			zap.Uint64("user_id", userID),
		)
	}
}

func (s *Service) cleanupExpiredMessages() {
	ctx := context.Background()
	currentTime := time.Now().Unix()
	
	// 清理过期的离线消息
	s.offlineQueuesMu.Lock()
	for userID, messages := range s.offlineQueues {
		var validMessages []*Message
		for _, msg := range messages {
			if msg.TTL == 0 || currentTime-msg.Timestamp < msg.TTL {
				validMessages = append(validMessages, msg)
			}
		}
		
		if len(validMessages) < len(messages) {
			s.offlineQueues[userID] = validMessages
			
			s.logger.Debug("清理过期离线消息",
				zap.Uint64("user_id", userID),
				zap.Int("removed_count", len(messages)-len(validMessages)),
			)
		}
	}
	s.offlineQueuesMu.Unlock()
}

func (s *Service) collectMetrics() {
	ticker := time.NewTicker(30 * time.Second)
	defer ticker.Stop()
	
	for range ticker.C {
		s.onlineUsersMu.RLock()
		onlineCount := len(s.onlineUsers)
		s.onlineUsersMu.RUnlock()
		
		s.offlineQueuesMu.RLock()
		offlineCount := 0
		for _, messages := range s.offlineQueues {
			offlineCount += len(messages)
		}
		s.offlineQueuesMu.RUnlock()
		
		s.metrics.OnlineUsers = int64(onlineCount)
		s.metrics.OfflineMessages = int64(offlineCount)
		
		s.logger.Debug("推送服务指标",
			zap.Int64("total_pushes", s.metrics.TotalPushes),
			zap.Int64("realtime_pushes", s.metrics.RealTimePushes),
			zap.Int64("offline_pushes", s.metrics.OfflinePushes),
			zap.Int64("online_users", s.metrics.OnlineUsers),
			zap.Int64("offline_messages", s.metrics.OfflineMessages),
			zap.Float64("avg_push_latency_ms", s.metrics.AvgPushLatency),
		)
	}
}

func (s *Service) recordPushLatency(latency time.Duration) {
	totalPushes := s.metrics.PushSuccess + s.metrics.PushFailed
	if totalPushes == 0 {
		s.metrics.AvgPushLatency = float64(latency.Milliseconds())
	} else {
		s.metrics.AvgPushLatency = (s.metrics.AvgPushLatency*float64(totalPushes-1) +
			float64(latency.Milliseconds())) / float64(totalPushes)
	}
}

func randomString(length int) string {
	const charset = "abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789"
	b := make([]byte, length)
	for i := range b {
		b[i] = charset[time.Now().UnixNano()%int64(len(charset))]
	}
	return string(b)
}

// WebSocketHandler WebSocket连接处理
func (h *Handler) WebSocketHandler(c *gin.Context) {
	// 获取用户ID
	userID, exists := c.Get("user_id")
	if !exists {
		c.JSON(http.StatusUnauthorized, gin.H{
			"code":    errors.CodeUnauthorized,
			"message": "用户未认证",
		})
		return
	}
	
	// 升级为WebSocket连接
	conn, err := h.pushService.WSManager().Upgrade(c.Writer, c.Request)
	if err != nil {
		h.logger.Error("升级WebSocket连接失败",
			zap.Error(err),
			zap.Uint64("user_id", userID.(uint64)),
		)
		c.JSON(http.StatusBadRequest, gin.H{
			"code":    errors.CodeInternalError,
			"message": "WebSocket连接失败",
		})
		return
	}
	
	// 创建用户会话
	session := &push.UserSession{
		UserID:       userID.(uint64),
		DeviceID:     c.GetHeader("X-Device-ID"),
		ClientType:   c.GetHeader("X-Client-Type"),
		LastSeen:     time.Now(),
		WsConnection: conn,
	}
	
	// 注册用户在线
	if err := h.pushService.RegisterUserOnline(userID.(uint64), session); err != nil {
		h.logger.Error("注册用户在线失败",
			zap.Error(err),
			zap.Uint64("user_id", userID.(uint64)),
		)
		conn.Close()
		return
	}
	
	// 处理WebSocket连接
	h.pushService.WSManager().HandleConnection(conn, userID.(uint64))
	
	// 连接关闭时注销用户
	defer func() {
		h.pushService.UnregisterUserOnline(userID.(uint64))
	}()
	
	// 保持连接
	h.pushService.WSManager().KeepAlive(conn)
}

	redisClient, err := database.InitRedis(cfg.Redis)
	if err != nil {
		logger.Fatal("初始化Redis失败", zap.Error(err))
	}
	
	// 初始化Kafka生产者
	kafkaProducer, err := kafka.NewProducer(&kafka.ProducerConfig{
		Brokers:        cfg.Kafka.Brokers,
		Topic:          cfg.Kafka.TaskTopic,
		MaxMessageSize: 1024 * 1024, // 1MB
		RequiredAcks:   -1,          // 等待所有副本确认
		Compression:    1,           // GZIP压缩
		MaxRetries:     3,
		RetryBackoff:   100 * time.Millisecond,
		BatchSize:      100,
		BatchTimeout:   100 * time.Millisecond,
	}, logger)
	if err != nil {
		logger.Fatal("初始化Kafka生产者失败", zap.Error(err))
	}
	
	// 初始化缓存
	cache := task.NewCache(redisClient, logger)
	
	// 初始化推送服务
	pushService := push.NewService(redisClient, cache, logger, cfg.Push)
	
	// 初始化任务分发器
	taskDispatcher := task.NewDispatcher(
		db,
		redisClient,
		kafkaProducer,
		cache,
		logger,
		cfg,
	)
	
	// 初始化API处理器
	apiHandler := api.NewHandler(taskDispatcher, pushService, logger)
	
	// 初始化Gin路由
	router := initRouter(apiHandler, cfg, logger)
	
	// 启动HTTP服务器
	server := &http.Server{
		Addr:         fmt.Sprintf(":%d", cfg.Server.Port),
		Handler:      router,
		ReadTimeout:  cfg.Server.ReadTimeout,
		WriteTimeout: cfg.Server.WriteTimeout,
		IdleTimeout:  cfg.Server.IdleTimeout,
	}
	
	// 启动健康检查服务器
	if cfg.Monitor.HealthCheckPort > 0 {
		go startHealthServer(cfg.Monitor.HealthCheckPort, logger)
	}
	
	// 启动指标服务器
	if cfg.Monitor.EnableMetrics && cfg.Monitor.MetricsPort > 0 {
		go startMetricsServer(cfg.Monitor.MetricsPort, logger)
	}
	
	// 优雅关机
	go func() {
		if err := server.ListenAndServe(); err != nil && err != http.ErrServerClosed {
			logger.Fatal("HTTP服务器启动失败", zap.Error(err))
		}
	}()
	
	logger.Info("API服务启动成功",

		zap.String("mode", cfg.Server.Mode),

	quit := make(chan os.Signal, 1)

	ctx, cancel := context.WithTimeout(context.Background(), 30*time.Second)
	defer cancel()
	
	// 停止HTTP服务器
	if err := server.Shutdown(ctx); err != nil {
		logger.Error("HTTP服务器关闭失败", zap.Error(err))
	}
	
	// 关闭Kafka生产者
	if err := kafkaProducer.Close(); err != nil {
		logger.Error("关闭Kafka生产者失败", zap.Error(err))
	}
	
	// 关闭Redis连接
	if err := redisClient.Close(); err != nil {
		logger.Error("关闭Redis连接失败", zap.Error(err))
	}
	
	// 关闭推送服务
	pushService.Close()
	
	// 关闭数据库连接
	if sqlDB, err := db.DB(); err == nil {
		if err := sqlDB.Close(); err != nil {
			logger.Error("关闭数据库连接失败", zap.Error(err))
		}
	}
	
	logger.Info("服务已优雅关闭")
}

func initLogger(cfg *config.Config) *zap.Logger {
	var logger *zap.Logger
	var err error
	
	if cfg.Log.Output == "stdout" {
		if cfg.Log.Format == "json" {
			logger, err = zap.NewProduction()
		} else {
			logger, err = zap.NewDevelopment()
		}
	} else {
		// 这里可以配置日志文件输出
		cfg := zap.NewProductionConfig()
		cfg.OutputPaths = []string{cfg.Log.Output}
		logger, err = cfg.Build()
	}
	
	if err != nil {
		log.Fatalf("初始化日志失败: %v", err)
	}
	
	return logger
}

func initRouter(handler *api.Handler, cfg *config.Config, logger *zap.Logger) *gin.Engine {
	// 设置Gin模式
	if cfg.Server.Mode == "release" {
		gin.SetMode(gin.ReleaseMode)
	}
	
	router := gin.New()
	
	// 中间件
	router.Use(gin.Recovery())
	router.Use(loggingMiddleware(logger))
	router.Use(corsMiddleware())
	router.Use(rateLimitMiddleware())
	router.Use(authMiddleware())
	
	// API路由
	v1 := router.Group("/api/v1")
	{
		// 健康检查
		v1.GET("/health", handler.HealthCheck)
		v1.GET("/metrics", handler.Metrics)
		
		// 任务相关
		tasks := v1.Group("/tasks")
		{
			tasks.POST("", handler.CreateTask)
			tasks.GET("", handler.ListTasks)
			tasks.GET("/:task_id", handler.GetTaskStatus)
			tasks.DELETE("/:task_id", handler.CancelTask)
		}
		
		// 模板相关
		v1.GET("/templates", handler.GetTemplates)
		
		// WebSocket
		v1.GET("/ws", handler.WebSocketHandler)
	}
	
	// 静态文件（用于展示生成的表情包）
	router.Static("/static", "./static")
	
	// 根路径重定向到API文档
	router.GET("/", func(c *gin.Context) {
		c.Redirect(http.StatusFound, "/docs")
	})
	
	return router
}

			zap.Int("status", c.Writer.Status()),
			zap.String("method", c.Request.Method),
			zap.String("path", path),
			zap.String("query", query),
			zap.String("ip", c.ClientIP()),
			zap.String("user-agent", c.Request.UserAgent()),
			zap.Duration("latency", latency),
			zap.Int("size", c.Writer.Size()),
		)
	}
}

func corsMiddleware() gin.HandlerFunc {
	return func(c *gin.Context) {
		c.Writer.Header().Set("Access-Control-Allow-Origin", "*")
		c.Writer.Header().Set("Access-Control-Allow-Credentials", "true")
		c.Writer.Header().Set("Access-Control-Allow-Headers", "Content-Type, Content-Length, Accept-Encoding, X-CSRF-Token, Authorization, accept, origin, Cache-Control, X-Requested-With")
		c.Writer.Header().Set("Access-Control-Allow-Methods", "POST, OPTIONS, GET, PUT, DELETE")
		
		if c.Request.Method == "OPTIONS" {
			c.AbortWithStatus(204)
			return
		}
		
		c.Next()
	}
}

func authMiddleware() gin.HandlerFunc {
	return func(c *gin.Context) {
		// 跳过健康检查和指标端点
		path := c.Request.URL.Path
		if path == "/api/v1/health" || path == "/api/v1/metrics" {
			c.Next()
			return
		}
		
		// 从Header获取Token
		token := c.GetHeader("Authorization")
		if token == "" {
			c.JSON(http.StatusUnauthorized, gin.H{
				"code":    errors.CodeUnauthorized,
				"message": "未提供认证令牌",
			})
			c.Abort()
			return
		}
		
		// 验证Token
		userID, err := utils.ValidateToken(token)
		if err != nil {
			c.JSON(http.StatusUnauthorized, gin.H{
				"code":    errors.CodeUnauthorized,
				"message": "认证令牌无效",
				"error":   err.Error(),
			})
			c.Abort()
			return
		}
		
		// 将用户ID存入上下文
		c.Set("user_id", userID)
		
		c.Next()
	}

## 7. 部署脚本

```bash
#!/bin/bash
# scripts/deploy.sh

set -e

ENV=${1:-production}
VERSION=${2:-latest}

echo "部署表情包合成系统到 $ENV 环境，版本: $VERSION"

# 构建镜像
echo "构建Docker镜像..."
docker build -f Dockerfile.api -t emoji-system/api-server:$VERSION .
docker build -f Dockerfile.task -t emoji-system/task-dispatcher:$VERSION .
docker build -f Dockerfile.consumer -t emoji-system/algorithm-consumer:$VERSION .
docker build -f Dockerfile.push -t emoji-system/push-service:$VERSION .

# 推送镜像到仓库
if [ "$ENV" = "production" ]; then
    echo "推送镜像到生产仓库..."
    docker tag emoji-system/api-server:$VERSION registry.example.com/emoji-system/api-server:$VERSION
    docker tag emoji-system/task-dispatcher:$VERSION registry.example.com/emoji-system/task-dispatcher:$VERSION
    docker tag emoji-system/algorithm-consumer:$VERSION registry.example.com/emoji-system/algorithm-consumer:$VERSION
    docker tag emoji-system/push-service:$VERSION registry.example.com/emoji-system/push-service:$VERSION
    
    docker push registry.example.com/emoji-system/api-server:$VERSION
    docker push registry.example.com/emoji-system/task-dispatcher:$VERSION
    docker push registry.example.com/emoji-system/algorithm-consumer:$VERSION
    docker push registry.example.com/emoji-system/push-service:$VERSION
fi

# 部署到Kubernetes
if [ "$ENV" = "production" ]; then
    echo "部署到Kubernetes生产集群..."
    kubectl apply -f deployments/k8s/namespace.yaml
    kubectl apply -f deployments/k8s/configmap.yaml
    kubectl apply -f deployments/k8s/secrets.yaml
    kubectl apply -f deployments/k8s/deployment.yaml
    kubectl apply -f deployments/k8s/service.yaml
    kubectl apply -f deployments/k8s/ingress.yaml
    
    # 滚动更新
    kubectl set image deployment/emoji-api-server api-server=registry.example.com/emoji-system/api-server:$VERSION -n emoji-system
    kubectl set image deployment/emoji-algorithm-consumer algorithm-consumer=registry.example.com/emoji-system/algorithm-consumer:$VERSION -n emoji-system
    
    # 等待部署完成
    echo "等待部署完成..."
    kubectl rollout status deployment/emoji-api-server -n emoji-system
    kubectl rollout status deployment/emoji-algorithm-consumer -n emoji-system
    
    echo "部署完成！"
else
    echo "使用Docker Compose启动开发环境..."
    docker-compose -f deployments/docker-compose.dev.yml up -d
    
    echo "开发环境启动完成！"
fi

# 健康检查
echo "执行健康检查..."
sleep 30
curl -f http://localhost:8080/api/v1/health || echo "健康检查失败"

echo "部署脚本执行完毕"
```
